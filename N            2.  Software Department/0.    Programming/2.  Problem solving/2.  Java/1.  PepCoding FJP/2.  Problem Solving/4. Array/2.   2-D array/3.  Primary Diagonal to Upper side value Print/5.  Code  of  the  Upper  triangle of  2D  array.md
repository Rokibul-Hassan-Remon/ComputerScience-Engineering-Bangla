        int m;

        int[][] a = new int[m][m];  
  
        for (int i = 0; i < m; i++)  
            for (int j = 0; j < m; j++)  
                a[i][j] = scn.nextInt();  
  
        for(int gap = 0 ; gap <m ; gap++){  
            for(int i = 0 , j= gap ; j < m ; j++,i++)  
            {  
                System.out.println(a[i][j]);  
            }  
        }  
    }  
}
